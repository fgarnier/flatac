method private set_locals(funinfos:Cil_types.fundec ) =
    List.iter (fun s ->
                 match s.vtype with
                     TInt (IInt,_) -> Hashtbl.add  v_int_locals s.vname ()
                   | _ -> ()                       
              ) funinfos.slocals;
    if Hashtbl.mem v_int_locals  "__retres" then Hashtbl.add v_int_return   "__retres" ()
